135        """

145    def training_step(self, batch, batch_id):
146        """
147        One training step in PytorchLightning.
148        You should not have to modify this.

179    def validation_step(self, batch, batch_id, split='val'):
180        """
181        One validation step in PytorchLightning.
182        You should not have to modify this.
183        For more information, see `pytorch_lightning.validation_step`.
184
185        A core difference here is that Wrangl will
186        1. compute metrics automatically
187        2. extract example contexts, predictions, and ground truths
188        3. save predictions to disk (and upload to S3, Wandb etc.) for debugging
189        """
190        feat = self.featurize(batch)
191        context = self.extract_context(feat, batch)
192        gold = self.extract_gold(feat, batch)
193
194        out = self.infer(feat, batch)
195        pred = self.extract_pred(out, feat, batch)
196
197        metrics = self.compute_metrics(pred, gold, batch)
198        for k, v in metrics.items():
199            self.log('{}_{}'.format(split, k), v, batch_size=self.infer_batch_size(batch))
200
201        # for logging to wandb
202        sample_ids = list(range(len(context)))
203        random.shuffle(sample_ids)
204        sample_ids = sample_ids[:self.hparams.val_sample_size]
205
206        random.shuffle(self.pred_samples)
207        for i in sample_ids:
208            self.pred_samples.insert(0, (context[i], pred[i], gold[i]))
209        self.pred_samples = self.pred_samples[:self.hparams.val_sample_size]
210
211        with open('pred_samples.json', 'wt') as f:
212            data = [dict(context=repr(context), gen=repr(gen), gold=repr(gold)) for context, gen, gold in self.pred_samples]
213            json.dump(data, f, indent=2)

216    def run_train_test(cls, cfg: OmegaConf, train_dataset: torch.utils.data.DataLoader, eval_dataset: torch.utils.data.DataLoader, model_kwargs=None):
217        """
218        Creates a model according to config and trains/evaluates it.
219
220        Args:
221            cfg: config to use.
222            train_dataset: dataset for training.
223            eval_dataset: dataset for validation.
224            model_kwargs: key ward arguments for model constructor.
225
226        This will additionally set up according to your config file
227            - manual seeds
228            - early stopping
229            - checkpoint saving
230            - gradient clipping
231            - callbacks (including an always on CSV logger)
232        """
233        model_kwargs = model_kwargs or {}
234        pl.utilities.seed.seed_everything(seed=cfg.seed, workers=True)
235        dout = os.getcwd()
236
237        logger = logging.getLogger(name='{}:train_test'.format(cls.__name__))
238        logger.info('Logging to {}'.format(dout))
239
240        checkpoint = C.ModelCheckpoint(
241            dirpath=dout,
242            monitor=cfg.early_stopping.monitor,
243            mode=cfg.early_stopping.mode,
244            filename='{epoch:02d}-{' + cfg.early_stopping.monitor + ':.4f}',
245            every_n_epochs=1,
246            save_top_k=3,
247            save_last=True,
248            verbose=True,
249        )
250
251        train_logger = [CSVLogger(save_dir=dout, name='logs', flush_logs_every_n_steps=cfg.flush_logs_every_n_steps)]
252        if cfg.wandb.enable:
253            train_logger.append(WandbLogger(project=cfg.wandb.project, name=cfg.wandb.name, entity=cfg.wandb.entity, save_dir=cfg.wandb.dir))
254
255        fconfig = os.path.join(dout, 'config.yaml')
256        OmegaConf.save(config=cfg, f=fconfig)
257
258        model = cls(cfg, **model_kwargs)
259
260        logger.info('Loading data')
261        if cfg.collate_fn == 'auto':
262            collate_fn = None
263        elif cfg.collate_fn == 'ignore':
264            def collate_fn(batch):
265                return batch
266        else:
267            collate_fn = cls.collate_fn
268        train_loader = DataLoader(train_dataset, batch_size=cfg.batch_size, collate_fn=collate_fn)
269        eval_loader = DataLoader(eval_dataset, batch_size=cfg.val_batch_size, collate_fn=collate_fn)
270
271        trainer = pl.Trainer(
272            precision=cfg.precision,
273            strategy=cfg.strategy,
274            gpus=cfg.gpus,
275            auto_lr_find=False,
276            auto_scale_batch_size=False,
277            auto_select_gpus=cfg.gpus > 0,
278            benchmark=True,
279            default_root_dir=dout,
280            gradient_clip_val=cfg.grad_clip_norm,
281            gradient_clip_algorithm='norm',
282            log_every_n_steps=cfg.log_every_n_steps,
283            val_check_interval=1 if cfg.debug else cfg.val_check_interval,
284            limit_val_batches=cfg.limit_val_batches,
285            max_steps=cfg.max_steps,
286            logger=train_logger,
287            callbacks=[checkpoint] + model.get_callbacks(),
288        )
289        if not cfg.test_only:
290            ckpt_path = None
291            if cfg.autoresume:
292                ckpt_path = os.path.join(dout, 'last.ckpt')
293                if not os.path.isfile(ckpt_path):
294                    ckpt_path = None
295            trainer.fit(model, train_loader, eval_loader, ckpt_path=ckpt_path)
296
297        model = cls.load_from_checkpoint(os.path.join(dout, 'last.ckpt'))
298        result = trainer.test(model, eval_loader, verbose=True, test=False)
299        with open(os.path.join(dout, 'eval_pred.json'), 'wt') as f:
300            json.dump(result, f, indent=2)
301        result = trainer.test(model, eval_loader, verbose=True, test=True)
302        with open(os.path.join(dout, 'eval_test.json'), 'wt') as f:
303            json.dump(result, f, indent=2)
304        logger.info('Finished!')

307    def run_inference(cls, cfg: OmegaConf, fcheckpoint: str, eval_dataset: torch.utils.data.DataLoader, model_kwargs=None, test=False):
308        """
309        Loads a model according to config and runs inference.
310
311        Args:
312            cfg: config to use.
313            fcheckpoint: model checkpoint to load.
314            eval_dataset: dataset for validation.
315            model_kwargs: key ward arguments for model constructor.
316            test: whether to evaluate metrics
317
318        Returns:
319            if `test==True` then returns a dictionary of scores, otherwise returns a list of model predictions.
320        """
321        model_kwargs = model_kwargs or {}
322        model = cls.load_from_checkpoint(fcheckpoint, **model_kwargs)
323
324        if cfg.collate_fn == 'auto':
325            collate_fn = None
326        elif cfg.collate_fn == 'ignore':
327            def collate_fn(batch):
328                return batch
329        else:
330            collate_fn = cls.collate_fn
331        eval_loader = DataLoader(eval_dataset, batch_size=cfg.batch_size, collate_fn=collate_fn)
332
333        trainer = pl.Trainer(
334            precision=cfg.precision,
335            strategy=cfg.strategy,
336            gpus=cfg.gpus,
337            auto_lr_find=False,
338            auto_scale_batch_size=False,
339            auto_select_gpus=cfg.gpus > 0,
340            benchmark=True,
341        )
342        if test:
343            result = trainer.test(model, eval_loader, verbose=True)
344        else:
345            result = []
346            for b in trainer.predict(model, eval_loader):
347                result.extend(b)
348        return result

350    @classmethod
351    def load_experiment(cls, dexp):
352        """
353        Loads experiment from disk.
354        """
355        import pandas as pd
356        dexp = pathlib.Path(dexp)
357        config = OmegaConf.to_container(OmegaConf.load(dexp.joinpath('config.yaml')), resolve=True)
358
359        # make sure collection exists for project
360        df = []
361        for flog in dexp.joinpath('logs').glob('*/*.csv'):
362            df.append(pd.read_csv(flog))
363        if df:
364            logs = pd.concat(df).to_dict()
365        else:
366            logs = {}
367        return config, logs
